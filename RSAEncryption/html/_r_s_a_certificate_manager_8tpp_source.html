54template <typename Format>
55RSACertificate<T> RSACertificate<T>::importCertificate(const std::string& data) {
56 std::string ownerName;
57 std::pair<T, T> publicKey;
58 T signature;
59
60 if constexpr (std::is_same_v<Format, PGPFormat>) {
61 std::string decoded = importPGPMessage(data);
62 size_t ownerEnd = decoded.find('\n');
63 ownerName = decoded.substr(0, ownerEnd);
64
65 size_t ePos = decoded.find("e=", ownerEnd) + 2;
66 size_t nPos = decoded.find("n=", ePos);
67 size_t sigPos = decoded.find("signature=", nPos);
68
69 publicKey.first = T(decoded.substr(ePos, nPos - ePos - 1));
70 publicKey.second = T(decoded.substr(nPos + 2, sigPos - nPos - 3));
71 signature = T(decoded.substr(sigPos + 10));
72
73 } else if constexpr (std::is_same_v<Format, JSONFormat>) {
74 auto start = data.find("{");
75 auto end = data.rfind("}");
76 if (start == std::string::npos || end == std::string::npos) {
77 throw std::invalid_argument("Invalid JSON certificate format");
78 }
79
80 auto ownerStart = data.find("\"ownerName\": \"", start) + 14;
81 auto ownerEnd = data.find("\"", ownerStart);
82 ownerName = data.substr(ownerStart, ownerEnd - ownerStart);
83
84 auto eStart = data.find("\"e\": \"", ownerEnd) + 6;
85 auto eEnd = data.find("\"", eStart);
86 publicKey.first = T(data.substr(eStart, eEnd - eStart));
87
88 auto nStart = data.find("\"n\": \"", eEnd) + 6;
89 auto nEnd = data.find("\"", nStart);
90 publicKey.second = T(data.substr(nStart, nEnd - nStart));
91
92 auto sigStart = data.find("\"signature\": \"", nEnd) + 14;
93 auto sigEnd = data.find("\"", sigStart);
94 signature = T(data.substr(sigStart, sigEnd - sigStart));
95 } else {
96 throw std::invalid_argument("Unsupported format for certificate import");
97 }
98
99 return RSACertificate<T>(ownerName, publicKey, signature, publicKey.second);
100}
